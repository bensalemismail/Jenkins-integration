package io.bensalem;

public class MathUtils {
	
	public int addition(int a,int b) {
		return a+b;
	}
	
	public int substraction(int a,int b) {
		return a-b;
	}
	
	public int division(int a,int b) {
		return a/b;
	}
	
	public int multiplication(int a,int b) {
		return a*b;
	}

}
